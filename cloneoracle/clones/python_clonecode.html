static Py_complex
c_acosh(Py_complex x)
{
 Py_complex z;
 z = c_sqrt(c_half);
 z = c_log(c_prod(z, c_sum(c_sqrt(c_sum(x,c_one)),
      c_sqrt(c_diff(x,c_one)))));
 return c_sum(z, z);
}
